// Main application initialization
document.addEventListener('DOMContentLoaded', function() {
    // Initialize map
    initMap();
    
    // Set up event listeners
    document.getElementById('generate-btn').addEventListener('click', convertCoordinates);
    document.getElementById('location-btn').addEventListener('click', locateMe);
    
    // Initialize with default location
    generateNKPINFromCoordinates(17.3850, 78.4867);
});

// Function to handle manual coordinate conversion
function convertCoordinates() {
    const coordsInput = document.getElementById('coordinates').value.trim();
    
    if (!coordsInput) {
        alert('Please enter coordinates in "lat, lng" format');
        return;
    }
    
    const coords = coordsInput.split(',').map(coord => parseFloat(coord.trim()));
    
    if (coords.length !== 2 || isNaN(coords[0]) || isNaN(coords[1])) {
        alert('Invalid coordinates. Please use format: "latitude, longitude"');
        return;
    }
    
    // Update map view and marker
    updateMapLocation(coords[0], coords[1]);
    
    // Update display and generate NK-PIN
    document.getElementById('display-coords').textContent = coordsInput;
    generateNKPINFromCoordinates(coords[0], coords[1]);
}

// Function to use current location
function locateMe() {
    if (!navigator.geolocation) {
        alert('Geolocation is not supported by your browser');
        return;
    }
    
    navigator.geolocation.getCurrentPosition(
        function(position) {
            const lat = position.coords.latitude;
            const lng = position.coords.longitude;
            
            // Update map view
            updateMapLocation(lat, lng);
            
            // Update inputs and generate NK-PIN
            const coordsText = `${lat.toFixed(6)}, ${lng.toFixed(6)}`;
            document.getElementById('coordinates').value = coordsText;
            document.getElementById('display-coords').textContent = coordsText;
            
            generateNKPINFromCoordinates(lat, lng);
        },
        function(error) {
            alert('Unable to retrieve your location. Please enable location services.');
            console.error('Geolocation error:', error);
        }
    );
}

// Function to generate NK-PIN from coordinates
async function generateNKPINFromCoordinates(lat, lng) {
    try {
        // Display loading state
        document.getElementById('nkpin-main-display').innerHTML = 
            '<i class="fas fa-spinner fa-spin"></i> Generating NK-PIN...';
        
        // Convert coordinates to DIGIPIN
        const digipin = await convertToDigipin(lat, lng);
        
        // Update DIGIPIN display
        document.getElementById('display-digipin').textContent = digipin;
        
        // Convert DIGIPIN to NK-PIN
        const nkpin = await convertToNKPIN(digipin);
        
        // Update NK-PIN displays
        document.getElementById('display-nkpin').textContent = nkpin;
        document.getElementById('nkpin-main-display').textContent = nkpin;
        
        // Add popup to marker
        addMarkerPopup(nkpin, lat, lng);
        
    } catch (error) {
        console.error('Error generating NK-PIN:', error);
        document.getElementById('nkpin-main-display').innerHTML = 
            '<span style="color: #ff6b6b;"><i class="fas fa-exclamation-circle"></i> Error generating NK-PIN</span>';
    }
}

// Map initialization and controls
let map = null;
let marker = null;
let currentLatLng = { lat: 17.3850, lng: 78.4867 }; // Default: Hyderabad

// Initialize the map
function initMap() {
    // Create map centered on India
    map = L.map('map').setView([20.5937, 78.9629], 5);
    
    // Add OpenStreetMap tile layer
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors',
        maxZoom: 19
    }).addTo(map);
    
    // Add click event to map
    map.on('click', function(e) {
        currentLatLng = e.latlng;
        
        // Update coordinates display
        document.getElementById('display-coords').textContent = 
            `${currentLatLng.lat.toFixed(6)}, ${currentLatLng.lng.toFixed(6)}`;
        document.getElementById('coordinates').value = 
            `${currentLatLng.lat.toFixed(6)}, ${currentLatLng.lng.toFixed(6)}`;
        
        // Update marker
        updateMarker(currentLatLng.lat, currentLatLng.lng);
        
        // Generate NK-PIN automatically on map click
        generateNKPINFromCoordinates(currentLatLng.lat, currentLatLng.lng);
    });
    
    // Initialize with a default marker
    marker = L.marker([currentLatLng.lat, currentLatLng.lng]).addTo(map);
}

// Update marker position
function updateMarker(lat, lng) {
    if (marker) {
        map.removeLayer(marker);
    }
    
    marker = L.marker([lat, lng]).addTo(map);
    map.setView([lat, lng], 15);
}

// Update map location
function updateMapLocation(lat, lng) {
    currentLatLng = { lat, lng };
    updateMarker(lat, lng);
}

// Add popup to marker
function addMarkerPopup(nkpin, lat, lng) {
    if (marker) {
        marker.bindPopup(`
            <div style="text-align: center;">
                <h4 style="margin: 0 0 8px 0; color: #1e3799;">NK-PIN Location</h4>
                <div style="font-size: 1.2rem; font-weight: bold; color: #4a69bd; margin-bottom: 5px;">
                    ${nkpin}
                </div>
                <div style="font-size: 0.9rem; color: #666;">
                    Coordinates: ${lat.toFixed(6)}, ${lng.toFixed(6)}
                </div>
                <button onclick="copyToClipboard('${nkpin}')" 
                        style="margin-top: 8px; padding: 5px 10px; background: #4a69bd; color: white; border: none; border-radius: 4px; cursor: pointer;">
                    Copy NK-PIN
                </button>
            </div>
        `).openPopup();
    }
}

// Copy NK-PIN to clipboard
function copyToClipboard(text) {
    navigator.clipboard.writeText(text).then(function() {
        alert('NK-PIN copied to clipboard: ' + text);
    }, function(err) {
        console.error('Could not copy text: ', err);
    });
}
